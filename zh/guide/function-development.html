<script setup>
import { ref, nextTick, onMounted } from 'vue';
import { cloneDeep } from 'lodash-es';
import usePager from '@/hooks/pager';
import column from './utils/column';
import { EditDialog, DetailDialog } from './dialog';
import { tableDataAssign } from '@/utils';
import { initiateApproveApi } from '@/api/laboratory/dispose'; // 不同业务接口可能不同
import { getCapacityDatesApi } from '@/api/asphalt/asphalt-analysis/index.js'; // 不同业务接口可能不同

import mockjs from 'mockjs';

import { useCustomI18n } from '@/locale';
const { t } = useCustomI18n();

const approvalRequest = { method: initiateApproveApi };

// 分页
let { loading, pagination, pageRequest } = usePager();

// 查询条件
const generateFormModel = () => ({
  projName: '',
  projType: '',
});
const formModel = ref(generateFormModel());

const timeRangePicker = ref(null);
// 表格
const columns = ref(
  column.map((val) => {
    return {
      ...val,
      title: t(val.title),
    };
  })
);

const tableData = ref([]);

let baseData = mockjs.mock({
  'records|30': [
    {
      key: '@id', // 主键
      orgName: '@pick(["1中队", "2中队", "3中队"])',
      workPlace: '@pick(["泰国", "新加坡", "印度尼西亚"])',
      workContent: '消灭怪兽',
      'validity|0-100': 0,
      'approvalStatus|0-3': 0,
      relatedApprovalForm: '@id',
      applicant: '@cname',
      applicantTime: () => {
        let now = new Date().getTime();
        let randomTime = Math.floor(Math.random() * 1000 * 60 * 60 * 24); // 生成一个随机的时间戳，单位为毫秒
        let date = new Date(now - randomTime);
        let isoString = date.toISOString();
        return `${isoString.slice(0, 10)} ${isoString.slice(11, 19)}`;
      },
    },
  ],
});

const listApi = async (params, data) => {
  const snapData = cloneDeep(baseData);
  console.log(data);

  let total = snapData.records.length;

  let start = (params.pageNumber - 1) * params.pageSize;
  let end = params.pageNumber * params.pageSize;
  snapData.records = snapData.records.slice(start, end);

  return {
    data: {
      records: snapData.records,
      total,
    },
  };
};

const getTableList = async () => {
  await nextTick();
  const timeRange = timeRangePicker.value.timeRange;
  const data = cloneDeep({
    startTime: timeRange[0],
    endTime: timeRange[1],
    ...formModel.value,
  });
  const list = await pageRequest({
    callback: listApi,
    params: {
      pageNumber: pagination.current,
      pageSize: pagination.pageSize,
    },
    data,
  });
  tableData.value = tableDataAssign(columns.value, list);
};
onMounted(() => {
  getTableList();
});

const handleDelete = (record) => {
  getTableList();
};

const handleQuery = () => {
  pagination.current = 1;
  getTableList();
};
const handleReset = () => {
  timeRangePicker.value.init();
  formModel.value = generateFormModel();
  handleQuery();
};
// 分页
const handlePageChange = async (val) => {
  pagination.current = val;
  await nextTick();
  getTableList();
};
// 条数
const handlePageSizeChange = async (val) => {
  pagination.pageSize = val;
  await nextTick();
  getTableList();
};
const onRefresh = () => {
  formModel.value = generateFormModel();
  pagination.current = 1;
  getTableList();
};
// 弹窗操作
const ApprovalDialogRef = ref(null),
  EditDialogRef = ref(null),
  DetailDialogRef = ref(null);
const handleDialog = async (name, row) => {
  await nextTick();
  name.init(row);
};
</script>

<script setup>
import { ref } from 'vue';
import { Message } from '@arco-design/web-vue';
import {
  addScheduleApi,
  editScheduleApi,
} from '@/api/comprehensive/images-progress/images-config';
import { cloneDeep } from 'lodash-es';
import { beforeUpload, customRequest } from '@/utils/upload'; // 封装了通用的上传方法
import { formModelAssign } from '@/utils';
import { useCustomI18n } from '@/locale';
const { t } = useCustomI18n();

// 基本设置
const modalVisible = ref(false);
const modalTitle = ref(t('新增弹窗'));
const isEdit = ref(false);
const pageLoading = ref(false);

const emits = defineEmits(['onRefresh']);

const fileList = ref([]);

// 定义表单元素和数据
const formRef = ref(null);
const generateFormModel = () => ({
  id: null,
  sectionId: '',
  workPlace: '',
  validity: null,
  workContent: '',
});

const formModel = ref(generateFormModel());

const rules = {
  sectionId: [{ required: true, message: t('请选择所属标段') }],
  workPlace: [{ required: true, message: t('请填写作业区域') }],
};
// 弹窗初始化
const init = async (row) => {
  modalVisible.value = true;
  isEdit.value = !!row;
  modalTitle.value = !row ? t('新增弹窗') : t('编辑弹窗'); // !!row 为 true 时，!row 为 false，!!row 为 true，!
  if (row) formModel.value = formModelAssign(formModel.value, row);
};
// 确认操作
const handleBeforeOk = async () => {
  const res = await formRef.value?.validate();
  if (!res) {
    try {
      const apiCall = isEdit.value ? editScheduleApi : addScheduleApi;
      const data = cloneDeep(formModel.value);
      await apiCall(data);
      Message.success(`${isEdit.value ? t('编辑成功') : t('新增成功')}`);
      emits('onRefresh');
      handleBeforeCancel();
    } catch (error) {
      console.log(error);
      return false;
    }
  } else {
    return false;
  }
};

// 取消操作
const handleBeforeCancel = () => {
  handleReset();
  modalVisible.value = false;
  return true;
};

// 重置表单
const handleReset = () => {
  formModel.value = generateFormModel();
};

// 暴露子组件 对象/方法
defineExpose({
  init,
});
</script>
